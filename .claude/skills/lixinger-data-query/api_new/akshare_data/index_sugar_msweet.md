|-----|-----|-----|
| -   | -   | -   |

| 名称   | 类型      | 描述  |
|------|---------|-----|
| 日期   | object  | -   |
| 综合价格 | float64 | -   |
| 原糖价格 | float64 | -   |
| 现货价格 | float64 | -   |

import akshare as ak
